CREATE TABLE dealers (
    dealer_id int4 GENERATED BY DEFAULT AS IDENTITY( INCREMENT BY 1 MINVALUE 1 MAXVALUE 2147483647 START 1 CACHE 10 NO CYCLE) NOT NULL PRIMARY KEY,
    dealer_name VARCHAR(255) NOT NULL,
    dealer_email VARCHAR(255) NOT NULL UNIQUE,
    subscription_type VARCHAR(50) NOT NULL
);

CREATE TABLE vehicles (
    id int4 GENERATED BY DEFAULT AS IDENTITY( INCREMENT BY 1 MINVALUE 1 MAXVALUE 2147483647 START 1 CACHE 10 NO CYCLE) NOT NULL PRIMARY KEY,
    model VARCHAR(255) NOT NULL,
    price DECIMAL(10, 2) NOT NULL,
    status VARCHAR(50) NOT NULL,
    dealer_id BIGINT NOT NULL,
    CONSTRAINT fk_dealer FOREIGN KEY (dealer_id) REFERENCES dealers(dealer_id) ON DELETE CASCADE
);
